日期

  • // 將日期 '2020/07/15' 轉換為 '2020年07月15日'
    const date = '2020/07/15'
    // 使用 group 把年、月、日抽取出來
    const regex = /(\d{4})\/(\d{2})\/(\d{2})/
    // 使用 replace + backreference 轉換字串 // 在 replace 中是使用 $1、$2、... 替代群組捕獲的文字
    date.replace(regex,'$1年$2月$3日') // "2020年07月15日"
